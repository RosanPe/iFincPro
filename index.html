Macro Google Apps Script (Código de Deploy)

Abaixo está o código que deve ser inserido no Editor de Scripts da sua Planilha Google. Ele escuta as conexões da API do iFinc Pro, salva os registros e responde as consultas em tempo real. 

// --- Cole este código no Google Apps Script ---
function doGet(e) {
  var action = e.parameter.action;
  if (action === 'pull') {
    return ContentService.createTextOutput(JSON.stringify({
      status: 'success',
      data: pullDatabase()
    })).setMimeType(ContentService.MimeType.JSON);
  }
  return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: 'Ação GET desconhecida' })).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var payload = JSON.parse(e.postData.contents);
    if (payload.action === 'sync') {
      processSyncQueue(payload.queue);
      return ContentService.createTextOutput(JSON.stringify({
        status: 'success',
        message: 'Sincronização concluída com sucesso!'
      })).setMimeType(ContentService.MimeType.JSON);
    }
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: err.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}

function pullDatabase() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  return {
    cartoes: getSheetRows(ss, "cartoes"),
    categorias: getSheetRows(ss, "categorias"),
    gastos: getSheetRows(ss, "gastos"),
    investimentos: getSheetRows(ss, "investimentos")
  };
}

function getSheetRows(ss, sheetName) {
  var sheet = getSheetNormalized(ss, sheetName);
  if (!sheet) return [];
  checkAndCreateHeaders(sheet, sheetName);
  var data = sheet.getDataRange().getValues();
  if (data.length <= 1) return [];
  var headers = data[0];
  var rows = [];
  for (var r = 1; r < data.length; r++) {
    var obj = {};
    for (var c = 0; c < headers.length; c++) {
      obj[headers[c]] = data[r][c];
    }
    rows.push(obj);
  }
  return rows;
}

function processSyncQueue(queue) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  for (var i = 0; i < queue.length; i++) {
    var trx = queue[i];
    var sheet = getSheetNormalized(ss, trx.type);
    if (!sheet) continue;
    
    checkAndCreateHeaders(sheet, trx.type);
    
    var op = trx.op;
    var data = trx.payload;
    
    if (op === 'CREATE') {
      appendRow(sheet, data, trx.type);
    } else if (op === 'UPDATE') {
      updateRow(sheet, data, trx.type);
    } else if (op === 'DELETE') {
      deleteRow(sheet, data, trx.type);
    }
  }
}

function appendRow(sheet, obj, type) {
  var headers = sheet.getDataRange().getValues()[0];
  var newRow = [];
  for (var i = 0; i < headers.length; i++) {
    var val = obj[mapHeaderName(headers[i], type)];
    newRow.push(val !== undefined ? val : "");
  }
  sheet.appendRow(newRow);
}

function updateRow(sheet, obj, type) {
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var idCol = headers.indexOf(type === 'categorias' ? 'nome' : 'id');
  if (idCol === -1) return;
  
  var targetId = obj[type === 'categorias' ? 'name' : 'id'];
  for (var r = 1; r < data.length; r++) {
    if (String(data[r][idCol]) === String(targetId)) {
      for (var c = 0; c < headers.length; c++) {
        var key = mapHeaderName(headers[c], type);
        if (obj[key] !== undefined) {
          sheet.getRange(r + 1, c + 1).setValue(obj[key]);
        }
      }
      break;
    }
  }
}

function deleteRow(sheet, obj, type) {
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var idCol = headers.indexOf(type === 'categorias' ? 'nome' : 'id');
  if (idCol === -1) return;
  
  var targetId = type === 'categorias' ? obj.nome : obj.id;
  for (var r = 1; r < data.length; r++) {
    if (String(data[r][idCol]) === String(targetId)) {
      sheet.deleteRow(r + 1);
      break;
    }
  }
}

function mapHeaderName(header, type) {
  if (type === 'cartoes') {
    if (header === 'id') return 'id';
    if (header === 'nome') return 'name';
    if (header === 'vencimentoDia') return 'dueDay';
  }
  if (type === 'categorias') {
    if (header === 'nome') return 'name';
  }
  if (type === 'gastos') {
    if (header === 'id') return 'id';
    if (header === 'data') return 'date';
    if (header === 'descricao') return 'description';
    if (header === 'valor') return 'value';
    if (header === 'categoria') return 'category';
    if (header === 'metodo') return 'method';
    if (header === 'cartaoId') return 'cardId';
    if (header === 'parcelas') return 'installments';
    if (header === 'emprestimo') return 'isLoan';
  }
  if (type === 'investimentos') {
    if (header === 'id') return 'id';
    if (header === 'data') return 'date';
    if (header === 'ativo') return 'asset';
    if (header === 'quantidade') return 'quantity';
    if (header === 'precoMedio') return 'averagePrice';
  }
  return header;
}

function getSheetNormalized(ss, targetName) {
  var sheets = ss.getSheets();
  var target = normalizeStr(targetName);
  for (var i = 0; i < sheets.length; i++) {
    var name = normalizeStr(sheets[i].getName());
    if (name === target) {
      return sheets[i];
    }
  }
  return null;
}

function normalizeStr(str) {
  if (!str) return "";
  return str.toString().toLowerCase()
            .replace(/[áàâãä]/g, "a")
            .replace(/[éèêë]/g, "e")
            .replace(/[íìîï]/g, "i")
            .replace(/[óòôõö]/g, "o")
            .replace(/[úùûü]/g, "u")
            .replace(/[ç]/g, "c")
            .trim();
}

function checkAndCreateHeaders(sheet, type) {
  var lastRow = sheet.getLastRow();
  var lastCol = sheet.getLastColumn();
  
  if (lastRow === 0 || (lastRow === 1 && lastCol === 1 && sheet.getRange(1, 1).getValue() === "")) {
    var headers = [];
    if (type === 'cartoes') headers = ['id', 'nome', 'vencimentoDia'];
    if (type === 'categorias') headers = ['nome'];
    if (type === 'gastos') headers = ['id', 'data', 'descricao', 'valor', 'categoria', 'metodo', 'cartaoId', 'parcelas', 'emprestimo'];
    if (type === 'investimentos') headers = ['id', 'data', 'ativo', 'quantidade', 'precoMedio'];
    
    if (headers.length > 0) {
      sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
      SpreadsheetApp.flush();
    }
  }
}
